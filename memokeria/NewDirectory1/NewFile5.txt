class MyBook : Book
    {
        private int price;
        public MyBook(string t, string a, int p) : base(t, a)
        {
            this.price = p;
        }

        public override void display()
        {
            Console.WriteLine($"Title: {this.title}");
            Console.WriteLine($"Author: {this.author}");
            Console.WriteLine($"Price: {this.price}");
        }
    }
