       class Node
        {
            public int data;
            public Node next;
            public Node(int d){
                data=d;
                next=null;
            }
        }
    
    public static  Node insert(Node head,int data)
        {
            Node hear = head;
            Node a = new Node(data);
            if( hear == null){
                hear = a;
            }else{
                while(hear.next != null)
                {
                    hear = hear.next;
                }
                hear.next = a;
                hear = head;
            }
        
            return hear;
            //Complete this method
        }

        public static void display(Node head)
        {
            Node start=head;
            while(start!=null)
            {
                Console.Write(start.data+" ");
                start=start.next;
            }
        }
